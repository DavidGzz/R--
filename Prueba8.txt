function int fibona(int a){
	int x, aux, aux1, sig;
	aux = 0;
	aux1 = 1;
	sig = 0;
	for(x = 0; x < a; x = x + 1;) {
		if(x == 1){
			write(aux);
		}
		if(x == 2) {
			write(aux1);
		}
		sig = aux + aux1;
		aux = aux1;
		aux1 = sig;
	}
	return(sig);
}

main(){
	int funcion;
	funcion = fibona(20);
	write(funcion);

}
